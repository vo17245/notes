## 相应类

相应类用于统一返回给前端的数据格式

```java
package vo17245.reggie.common;

import lombok.Data;

import java.util.HashMap;
import java.util.Map;

@Data
public class Respone<T> {
    private Integer code; //编码：1成功，0和其它数字为失败

    private String msg; //错误信息

    private T data; //数据

    private Map map = new HashMap(); //动态数据

    public static <T> Respone<T> success(T object) {
        Respone<T> r = new Respone<T>();
        r.data = object;
        r.code = 1;
        return r;
    }

    public static <T> Respone<T> error(String msg) {
        Respone r = new Respone();
        r.msg = msg;
        r.code = 0;
        return r;
    }

    public Respone<T> add(String key, Object value) {
        this.map.put(key, value);
        return this;
    }
}

```
